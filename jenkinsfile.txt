pipeline {
    agent any
    parameters {
        string(name: 'SOLUTION', description: 'Solution file name to build')
	string(name: 'DLL', description: 'dll file path')
	string(name: 'IMAGE_NAME', description: 'docker image name')
	string(name: 'PORT_NUMBER', description: 'port number to run')
    }
    stages {
        stage('Build') {
            steps {
                bat "dotnet build %params.SOLUTION% -p:configuration=release -v:n"
                echo "Building......."
            }
        }
	stage('Test') {
	    steps {
		bat 'dotnet test'
	    }
	}
	stage('Publish') {
	    steps {
			bat 'dotnet publish'
	    }
	}
     }
     post {
	success {
		archiveArtifacts artifacts: '**', fingerprint: true
		bat "docker build -t %IMAGE_NAME% ."
		bat "docker run -p %PORT_NUMBER%:80 %IMAGE_NAME%"
	}
    }
}
