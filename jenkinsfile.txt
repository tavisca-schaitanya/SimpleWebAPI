pipeline {
    agent any
    parameters {
	string(name: 'PROJECT_NAME')
    }
    stages {
        stage('Build') {
            steps {
                powershell(script: 'dotnet build')
           
            }
        }
	stage('Test') {
	    steps {
		powershell(script: 'dotnet test')
	    }
	}
	stage('Publish') {
	    steps {
		powershell(script: 'dotnet publish')
	    }
	}
	stage('Archive') {
	    steps {
		archiveArtifacts artifacts: '${env:PROJECT_NAME}/bin/Debug/netcoreapp2.1/publish/*.*', fingerprint: true
	    }
	}
     }
     post {
	  success {
	    	powershell(script: 'docker build -t image1 .')
	  }
    }
}
