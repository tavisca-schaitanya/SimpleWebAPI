pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                powershell(script: 'dotnet build')
           
            }
        }
	stage('Test') {
	    steps {
		powershell(script: 'dotnet test')
	    }
	}
	stage('Publish') {
	    steps {
		powershell(script: 'dotnet publish')
	    }
	}
	stage('Archive') {
	    steps {
		archiveArtifacts artifacts: './bin/Debug/netcoreapp2.1/publish/*', fingerprint: true
	    }
	}
     }
}
