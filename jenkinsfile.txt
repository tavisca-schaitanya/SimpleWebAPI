pipeline {
    agent any
    parameters {
        string(name: 'SOLUTION', description: 'Solution file name to build')
	string(name: 'DLL', description: 'Dll file name to run')
    }
    stages {
        stage('Build') {
            steps {
                bat 'dotnet build ${params.SOLUTION} -p:configuration=release -v:n'
                echo "Building......."
            }
        }
	stage('Test') {
	    steps {
		bat 'dotnet test'
	    }
	}
	stage('Publish') {
	    steps {
			bat 'dotnet publish'
	    }
	}
     }
     post {
	success {
		archiveArtifacts artifacts: '**', fingerprint: true
		bat 'dotnet ${params.DLL}'
	}
    }
}
